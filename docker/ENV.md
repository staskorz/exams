# .env file format

```dotenv
#.env
COMPOSE_PROJECT_NAME=DOCKER_COMPOSE_PROJECT_NAME
IP_ADDRESS=IP_ADDRESS_TO_BIND_WEB_SERVER_TO
AD_URL=ldap://AD_SERVER_HOSTNAME
AD_BASE_DN=AD_BASE_DN_TO_LOOKUP_USERS_FROM
AD_USERNAME=AD_USERNAME_FOR_AD_QUERIES
AD_PASSWORD=AD_USER_PASSWORD
AD_GROUP_OPERATORS=AD_GROUP_CONTAINING_APPLICATION_ADMINS
HTTP_PORT=HTTP_PORT_TO_BIND_WEB_SERVER_TO
MONGODB_URL=mongodb://MONGODB_HOST:MONGODB_PORT
```
